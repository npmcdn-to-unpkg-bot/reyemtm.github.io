  var gray = L.tileLayer('https://cartodb-basemaps-{s}.global.ssl.fastly.net/light_all/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap © CartoDB',
      subdomains: 'abcd',
      maxZoom: 19
  }).addTo(map);

  var fire = L.tileLayer('https://{s}.tile.thunderforest.com/spinal-map/{z}/{x}/{y}.png', {
      attribution: '© Thunderforest, © OpenStreetMap'
  });

  var pioneer = L.tileLayer('https://{s}.tile.thunderforest.com/pioneer/{z}/{x}/{y}.png', {
      attribution: '© Thunderforest, © OpenStreetMap'
  });

  var basemaps = {
    "Gray": gray,
    "Fire": fire,
    "Pioneer": pioneer,
  };

  var layerControl = new L.control.layers(basemaps).addTo(map);

    Choropleth Made Simple

    Choropleth Plugin

    
    L.choropleth(counties, {
        valueProperty: 'pci', // which property in the features to use
        scale: ['white', 'red'], // chroma.js scale - include as many as you like
        steps: 5, // number of breaks or steps in range
        mode: 'q', // q for quantile, e for equidistant, k for k-means
        style: {
            color: '#fff', // border color
            weight: 2,
            fillOpacity: 0.8
        },
        onEachFeature: function(feature, layer) {
            layer.bindPopup(feature.properties.value)
        }
    }).addTo(map)
